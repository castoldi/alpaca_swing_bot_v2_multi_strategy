# SMA 50 Cross Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add a stop-protected, long-only SMA(50) price-cross strategy that trades completed daily candles while every existing strategy remains on 4-hour candles.

**Architecture:** Strategy metadata selects its timeframe and exit mode. The shared feed fetches either daily or 4-hour bars, the backtest engine has a dedicated next-session lifecycle for signal exits, and the live bot uses an Alpaca stop-only OTO entry plus an ownership-protected crossover exit. Registry metadata drives reports and dashboard cards.

**Tech Stack:** Python 3.11, pandas, alpaca-py, FastAPI, SQLite, Plotly, pytest, PowerShell process manager.

## Global Constraints

- Strategy key: `sma_50_cross`; label: `SMA 50 Cross`.
- Entry: completed daily close crosses from `<= SMA(50)` to `> SMA(50)`.
- Normal exit: completed daily close crosses from `>= SMA(50)` to `< SMA(50)`.
- Risk exit: broker-held stop 10% below the fresh live market reference or backtest entry open.
- No take profit, time stop, short position, repeated above-SMA entry, or incomplete daily candle.
- Existing six strategies remain on `4h` bars with their current behavior.
- Live trading stays paper-only and whole-share sizing stays `$200/trade` with a safe skip below one share.
- Bot and dashboard starts/restarts use only `pwsh scripts/manage.ps1`.
- Do not start either service from a linked worktree; restart from the canonical main checkout only after merge, preventing a second project process.
- Every implementation commit is documented, tagged, and pushed.

```python
from __future__ import annotations
from typing import Optional

import numpy as np
import pandas as pd

from config import PARAMS, StrategyParams
from .base import BaseStrategy, EntrySignal


class SMA50CrossStrategy(BaseStrategy):
    name = "sma_50_cross"
    label = "SMA 50 Cross"
    version = "V2"
    color = "#38bdf8"
    timeframe = "1d"
    exit_mode = "signal_with_stop"
    has_take_profit = False
    description = (
        "Daily close crosses above SMA(50) to buy and crosses below to sell. "
        "A broker-held 10% emergency stop protects the position."
    )
    params_display = ["Daily", "SMA 50", "SL 10%", "Cross-down exit"]

    @staticmethod
    def _values(df: pd.DataFrame, idx: int, p: StrategyParams):
        if idx < p.sma_cross_period:
            return None
        prev, row = df.iloc[idx - 1], df.iloc[idx]
        values = (prev["close"], prev["sma_cross"], row["close"], row["sma_cross"])
        return values if all(np.isfinite(float(v)) for v in values) else None

    def check_entry(self, df: pd.DataFrame, idx: int,
                    p: StrategyParams = PARAMS) -> Optional[EntrySignal]:
        values = self._values(df, idx, p)
        if values is None:
            return None
        prev_close, prev_sma, close, current_sma = values
        if not (prev_close <= prev_sma and close > current_sma):
            return None
        row = df.iloc[idx]
        entry = float(close)
        return EntrySignal(
            date=pd.Timestamp(row.name), entry_price=entry,
            stop_loss=entry * (1.0 - p.sma_cross_stop_loss_pct),
            take_profit=0.0, atr=float(row["atr"]), rsi=float(row["rsi"]),
            strategy=self.name,
        )

    def check_exit(self, df: pd.DataFrame, idx: int,
                   p: StrategyParams = PARAMS) -> Optional[str]:
        values = self._values(df, idx, p)
        if values is None:
            return None
        prev_close, prev_sma, close, current_sma = values
        return "sma_cross_down" if prev_close >= prev_sma and close < current_sma else None
```

Register `SMA50CrossStrategy()` after `RegimeAdaptiveStrategy()` and before `EnsembleStrategy()`.

```python
from zoneinfo import ZoneInfo

_ET = ZoneInfo("America/New_York")


def _timeframe_parts(timeframe: str) -> tuple[int, str]:
    if timeframe == "4h":
        return 4, "Hour"
    if timeframe == "1d":
        return 1, "Day"
    raise ValueError(f"Unsupported timeframe: {timeframe}")


def _feed_options() -> dict:
    try:
        from alpaca.data.enums import Adjustment, DataFeed
        return {"feed": DataFeed.IEX, "adjustment": Adjustment.ALL}
    except Exception:
        return {}


def fetch_bars(ticker, start, end, timeframe=BAR_TIMEFRAME):
    try:
        amount, unit_name = _timeframe_parts(timeframe)
        from alpaca.data.requests import StockBarsRequest
        from alpaca.data.timeframe import TimeFrame, TimeFrameUnit
        unit = getattr(TimeFrameUnit, unit_name)
        request = StockBarsRequest(
            symbol_or_symbols=ticker,
            timeframe=TimeFrame(amount, unit),
            start=_as_dt(start), end=_as_dt(end), **_feed_options(),
        )
        return _normalize(_get_client().get_stock_bars(request).df, ticker)
    except Exception as exc:
        log.warning("fetch_bars(%s, %s) failed: %s", ticker, timeframe, exc)
        return pd.DataFrame(columns=_OHLCV)


def fetch_recent(ticker: str, days: int = 120,
                 timeframe: str = BAR_TIMEFRAME) -> pd.DataFrame:
    end = datetime.now(timezone.utc)
    return fetch_bars(ticker, end - timedelta(days=days), end, timeframe)


def completed_bars(df: pd.DataFrame, timeframe: str,
                   as_of: datetime | None = None) -> pd.DataFrame:
    if timeframe != "1d" or df.empty:
        return df
    current = as_of or datetime.now(_ET)
    if current.tzinfo is None:
        current = current.replace(tzinfo=_ET)
    session_date = current.astimezone(_ET).date()
    return df[pd.Index(df.index.date) < session_date].copy()
```
